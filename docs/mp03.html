load_songs <- function() {
  library(readr)
  library(dplyr)
  library(tidyr)
  library(stringr)

  # Define directory, file path, and URL
  dir_path <- "data/mp03"
  file_path <- file.path(dir_path, "songs.csv")
  url <- "https://raw.githubusercontent.com/gabminamedez/spotify-data/refs/heads/master/data.csv"

  # Create directory if it doesn't exist
  if (!dir.exists(dir_path)) {
    dir.create(dir_path, recursive = TRUE)
  }

  # Download the file if it doesn't exist
  if (!file.exists(file_path)) {
    download.file(url, file_path, method = "libcurl")
  }

  # Read the CSV
  SONGS <- read_csv(file_path, show_col_types = FALSE)

  # Clean and split artist list
  SONGS_clean <- SONGS |>
    mutate(
      artists = str_remove_all(artists, "\\[|\\]|'")  # remove brackets and quotes
    ) |>
    separate_rows(artists, sep = ",\\s*") |>           # split on comma + optional space
    rename(artist = artists)                           # rename for clarity

  return(SONGS_clean)
}

# Define function to load playlists and return the data as a data frame
load_playlists <- function() {
  # First, ensure that all necessary files are downloaded
  download_files()
  
  # List the downloaded JSON files in the local directory
  file_list <- list.files(path = "spotify_data", pattern = "*.json", full.names = TRUE)
  
  # Initialize an empty list to store the playlist data
  playlist_data <- list()
  
  # Loop through each JSON file and read the data
  for (file in file_list) {
    print(paste("Reading file:", file))
    
    # Check if the file is a valid JSON file
    tryCatch({
      raw_data <- fromJSON(file)
      
      # Check the structure of the raw data
      if (!is.null(raw_data$playlists)) {
        # Process playlists
        for (playlist in raw_data$playlists) {
          
          # Ensure 'tracks' exists and is a list
          if (is.list(playlist$tracks) && length(playlist$tracks) > 0) {
            playlist_name <- playlist$name
            playlist_id <- playlist$pid
            playlist_followers <- playlist$num_followers
            playlist_position <- 0  # Adjust based on the playlist data (or remove it)
            
            # Iterate through the tracks within each playlist
            for (track in playlist$tracks) {
              artist_name <- track$artist_nam
              track_name <- track$track_name
              track_id <- track$track_uri 
              album_name <- track$album_name
              album_id <- track$album_uri 
              duration <- track$duration_ms / 1000  # Convert duration from ms to seconds
              
              # Append each track to the data frame
              playlist_data <- append(playlist_data, list(data.frame(
                playlist_name = playlist_name,
                playlist_id = playlist_id,
                playlist_position = playlist_position,
                playlist_followers = playlist_followers,
                artist_name = artist_name,
                track_name = track_name,
                track_id = track_id,
                album_name = album_name,
                album_id = album_id,
                duration = duration
              )))
              
              playlist_position <- playlist_position + 1
            }
          } else {
            print(paste("No tracks found for playlist:", playlist$name))
          }
        }
      } else {
        print("No playlists found in this file.")
      }
    }, error = function(e) {
      print(paste("Error reading file:", file, "- Possible non-JSON content"))
    })
  }
  
  # Combine all the individual data frames into one
  if (length(playlist_data) > 0) {
    playlist_df <- bind_rows(playlist_data)
    return(playlist_df)
  } else {
    return(tibble())  # Return an empty tibble if no data
  }
}
